# CMakeLists.txt Simplificado para Teste
cmake_minimum_required(VERSION 3.15)
project(analise_hash LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Arquivos do projeto
set(SOURCES
    src/main.cpp
    src/TabelaHashEncadeada.cpp
    src/TabelaHashAberta.cpp
    src/GeradorDados.cpp
)

include_directories(include)

# Criar executável
add_executable(analise_hash ${SOURCES})

# Link com threads se disponível
find_package(Threads)
if(Threads_FOUND)
    target_link_libraries(analise_hash Threads::Threads)
endif()

# Mensagens de debug
message(STATUS "Compilador: ${CMAKE_CXX_COMPILER_ID}")
message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")
